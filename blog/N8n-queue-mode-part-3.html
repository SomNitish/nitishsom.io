Load Testing Script: Simulating Alert Storm

We created a Python script that fires N concurrent alerts to the N8N webhook, measuring latency and verifying all reach the database.

The Load Testing Script

#!/usr/bin/env python3
"""
N8N Queue Mode Load Tester
Sends concurrent Wazuh-formatted alerts to N8N webhook
Measures latency, throughput, and database consistency
"""

import requests
import json
import time
import threading
from datetime import datetime
from collections import defaultdict
import sys

# Configuration
WEBHOOK_URL = "https://n8n-server:5678/webhook/[webhook-id]"
NUM_ALERTS = 500
CONCURRENT_THREADS = 50
TEST_DURATION_SECONDS = 60

# Metrics
metrics = {
    'sent': 0,
    'success': 0,
    'failed': 0,
    'latencies': [],
    'errors': defaultdict(int),
    'lock': threading.Lock()
}

def generate_wazuh_alert(alert_num):
    """Generate a fake Wazuh alert JSON"""
    source_ips = [
        "203.0.113.42",      # External IP (will trigger unauthorized)
        "198.51.100.55",     # Another external
        "192.0.2.100",       # Another external
        "172.16.0.50",       # Another external
    ]
    
    users = [f"test_user_{alert_num % 10}"]
    endpoints = [f"linux-server-{alert_num % 5}"]
    
    return {
        "body": {
            "all_fields": {
                "timestamp": datetime.utcnow().isoformat() + "Z",
                "rule": {
                    "level": 3,
                    "description": "sshd: authentication success.",
                    "id": "5715"
                },
                "agent": {
                    "id": "001",
                    "name": endpoints[0]
                },
                "data": {
                    "srcip": source_ips[alert_num % len(source_ips)],
                    "srcport": str(30000 + alert_num),
                    "dstuser": users[0]
                },
                "location": "auth.log"
            }
        }
    }

def send_alert(alert_num):
    """Send a single alert to N8N and record metrics"""
    try:
        payload = generate_wazuh_alert(alert_num)
        start_time = time.time()
        
        response = requests.post(
            WEBHOOK_URL,
            json=payload,
            verify=False,  # Self-signed cert
            timeout=10
        )
        
        latency = (time.time() - start_time) * 1000  # ms
        
        with metrics['lock']:
            metrics['sent'] += 1
            metrics['latencies'].append(latency)
            
            if response.status_code == 200:
                metrics['success'] += 1
            else:
                metrics['failed'] += 1
                metrics['errors'][f"HTTP {response.status_code}"] += 1
                
        if metrics['sent'] % 50 == 0:
            print(f"[+] Sent {metrics['sent']} alerts...")
            
    except Exception as e:
        with metrics['lock']:
            metrics['failed'] += 1
            metrics['errors'][str(type(e).__name__)] += 1
        print(f"[-] Error on alert {alert_num}: {e}")

def run_load_test():
    """Execute the load test with concurrent threads"""
    print(f"Starting load test: {NUM_ALERTS} alerts, {CONCURRENT_THREADS} threads")
    print(f"Webhook: {WEBHOOK_URL}")
    print()
    
    threads = []
    start_time = time.time()
    
    # Launch threads
    for i in range(NUM_ALERTS):
        thread = threading.Thread(target=send_alert, args=(i,))
        threads.append(thread)
        thread.start()
        
        # Rate limit: spread threads over the duration
        if i % CONCURRENT_THREADS == 0:
            time.sleep(0.1)
    
    # Wait for all threads
    for thread in threads:
        thread.join()
    
    elapsed = time.time() - start_time
    
    # Results
    print("\n" + "="*60)
    print("LOAD TEST RESULTS")
    print("="*60)
    print(f"Total Time:        {elapsed:.2f}s")
    print(f"Alerts Sent:       {metrics['sent']}")
    print(f"Successful:        {metrics['success']} ({100*metrics['success']/metrics['sent']:.1f}%)")
    print(f"Failed:            {metrics['failed']} ({100*metrics['failed']/metrics['sent']:.1f}%)")
    print(f"Throughput:        {metrics['success']/elapsed:.1f} alerts/sec")
    print()
    
    if metrics['latencies']:
        latencies = sorted(metrics['latencies'])
        print(f"Latency (min):     {min(latencies):.2f}ms")
        print(f"Latency (avg):     {sum(latencies)/len(latencies):.2f}ms")
        print(f"Latency (p95):     {latencies[int(len(latencies)*0.95)]:.2f}ms")
        print(f"Latency (max):     {max(latencies):.2f}ms")
    print()
    
    if metrics['errors']:
        print("Errors by Type:")
        for error_type, count in sorted(metrics['errors'].items(), key=lambda x: -x[1]):
            print(f"  {error_type}: {count}")
    
    print("="*60)
    print(f"\n✅ Load test complete. Check N8N and Postgres for data.")

if __name__ == "__main__":
    try:
        run_load_test()
    except KeyboardInterrupt:
        print("\n\n[!] Test interrupted by user")
        sys.exit(1)

Regression Testing: Verify Data Consistency

Load test tells you if your system survives stress. Regression tests tell you if data is correct under stress.

Regression Test Script

This script sends alerts, then queries Postgres to verify:

  1. All 500 incidents were created
  2. All incident IDs are unique (no collisions)
  3. No data is missing or truncated
  4. All unauthorized alerts created Jira tickets
#!/usr/bin/env python3
"""
N8N Queue Mode Regression Test
Validates data consistency after load test
"""

import psycopg2
from psycopg2.extras import RealDictCursor
import sys
import time

# Configuration
DB_HOST = "n8n-server"
DB_PORT = 5433
DB_NAME = "n8n_queue"
DB_USER = "n8n_queue"
DB_PASSWORD = "your_password_here"

def connect_db():
    """Connect to PostgreSQL"""
    try:
        conn = psycopg2.connect(
            host=DB_HOST,
            port=DB_PORT,
            database=DB_NAME,
            user=DB_USER,
            password=DB_PASSWORD
        )
        return conn
    except Exception as e:
        print(f"[-] Database connection failed: {e}")
        sys.exit(1)

def run_regression_tests():
    """Execute all regression tests"""
    conn = connect_db()
    cursor = conn.cursor(cursor_factory=RealDictCursor)
    
    print("Running Regression Tests...")
    print("=" * 60)
    
    # Test 1: Count incidents
    cursor.execute("SELECT COUNT(*) as count FROM incidents;")
    result = cursor.fetchone()
    incident_count = result['count']
    print(f"\n[Test 1] Total Incidents Created")
    print(f"  Expected: >= 500")
    print(f"  Actual:   {incident_count}")
    print(f"  Status:   {'✅ PASS' if incident_count >= 500 else '❌ FAIL'}")
    
    # Test 2: Check for duplicate incident IDs
    cursor.execute("""
        SELECT incident_id, COUNT(*) as cnt 
        FROM incidents 
        GROUP BY incident_id 
        HAVING COUNT(*) > 1;
    """)
    duplicates = cursor.fetchall()
    print(f"\n[Test 2] Unique Incident IDs")
    print(f"  Expected: 0 duplicates")
    print(f"  Actual:   {len(duplicates)} duplicates")
    print(f"  Status:   {'✅ PASS' if len(duplicates) == 0 else '❌ FAIL'}")
    if duplicates:
        for dup in duplicates:
            print(f"    - {dup['incident_id']}: {dup['cnt']} occurrences")
    
    # Test 3: Check for NULL values
    cursor.execute("""
        SELECT 
            COUNT(*) filter (WHERE incident_id IS NULL) as null_ids,
            COUNT(*) filter (WHERE alert_name IS NULL) as null_names,
            COUNT(*) filter (WHERE severity IS NULL) as null_severity
        FROM incidents;
    """)
    result = cursor.fetchone()
    null_count = sum([
        result['null_ids'] or 0,
        result['null_names'] or 0,
        result['null_severity'] or 0
    ])
    print(f"\n[Test 3] Data Completeness (No NULLs)")
    print(f"  Expected: 0 NULL values")
    print(f"  Actual:   {null_count} NULL values")
    print(f"  Status:   {'✅ PASS' if null_count == 0 else '❌ FAIL'}")
    
    # Test 4: Check UNAUTHORIZED vs AUTHORIZED split
    cursor.execute("""
        SELECT 
            severity,
            COUNT(*) as count
        FROM incidents
        GROUP BY severity
        ORDER BY severity;
    """)
    results = cursor.fetchall()
    print(f"\n[Test 4] Incident Severity Distribution")
    for row in results:
        print(f"  {row['severity']}: {row['count']} incidents")
    print(f"  Status:   ✅ PASS (distribution looks reasonable)")
    
    # Test 5: Check Jira linkage
    cursor.execute("""
        SELECT 
            COUNT(*) as total,
            COUNT(*) filter (WHERE jira_key IS NOT NULL) as with_jira
        FROM incidents
        WHERE severity = 'HIGH';
    """)
    result = cursor.fetchone()
    total_high = result['total']
    with_jira = result['with_jira']
    print(f"\n[Test 5] Jira Ticket Creation")
    print(f"  HIGH severity incidents: {total_high}")
    print(f"  With Jira tickets:       {with_jira}")
    print(f"  Coverage:                {100*with_jira/total_high:.1f}%")
    print(f"  Status:   {'✅ PASS' if with_jira/total_high > 0.95 else '⚠️ WARNING'}")
    
    # Test 6: Sequence integrity
    cursor.execute("""
        SELECT 
            MIN(seq) as min_seq,
            MAX(seq) as max_seq,
            COUNT(*) as count
        FROM incidents;
    """)
    result = cursor.fetchone()
    min_seq = result['min_seq']
    max_seq = result['max_seq']
    actual_count = result['count']
    expected_count = max_seq - min_seq + 1
    
    print(f"\n[Test 6] Sequence Integrity")
    print(f"  Sequence range: {min_seq} to {max_seq}")
    print(f"  Expected gaps:  0")
    print(f"  Actual count:   {actual_count} (expected {expected_count})")
    print(f"  Status:   {'✅ PASS' if actual_count == expected_count else '❌ FAIL'}")
    
    print("\n" + "=" * 60)
    print("Regression Tests Complete")
    print("=" * 60)
    
    cursor.close()
    conn.close()

if __name__ == "__main__":
    run_regression_tests()

Auto-Scaling: Dynamic Worker Management

With 2 workers, you handle normal load. But what happens when 500 alerts arrive in 60 seconds? You need to scale to 6 workers automatically.

The Auto-Scaler Script

This daemon monitors Redis queue depth and scales workers up/down:

#!/bin/bash

# N8N Auto-Scaler Daemon
# Monitors Redis queue depth and scales workers dynamically
# Run as: systemd service or screen session

PROJECT_DIR="/opt/n8n-queue-mode"
MIN_WORKERS=2
MAX_WORKERS=6
SCALE_UP_THRESHOLD=15
SCALE_DOWN_THRESHOLD=3
CHECK_INTERVAL=10
COOLDOWN=60

cd "$PROJECT_DIR" || exit 1

REDIS_PASSWORD=$(grep ^REDIS_PASSWORD .env | cut -d= -f2)
LAST_SCALE_TIME=0

log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1"
}

get_queue_depth() {
    docker exec n8nq-redis redis-cli -a "$REDIS_PASSWORD" --no-auth-warning \
        LLEN bull:jobs:wait 2>/dev/null || echo 0
}

get_active_jobs() {
    docker exec n8nq-redis redis-cli -a "$REDIS_PASSWORD" --no-auth-warning \
        SCARD bull:jobs:active 2>/dev/null || echo 0
}

get_current_workers() {
    docker compose ps n8nq-worker --format json 2>/dev/null | jq -s 'length'
}

scale_workers() {
    local target=$1
    log "Scaling to $target workers..."
    docker compose up -d --scale n8nq-worker="$target" --no-recreate
    if [ $? -eq 0 ]; then
        log "✅ Scaled to $target workers"
    else
        log "❌ Failed to scale to $target workers"
    fi
}

echo "N8N Auto-Scaler Started"
echo "Min: $MIN_WORKERS | Max: $MAX_WORKERS"
echo "Up Threshold: $SCALE_UP_THRESHOLD | Down Threshold: $SCALE_DOWN_THRESHOLD"
echo

while true; do
    PENDING=$(get_queue_depth)
    ACTIVE=$(get_active_jobs)
    CURRENT=$(get_current_workers)
    TOTAL_LOAD=$((PENDING + ACTIVE))
    CURRENT_TIME=$(date +%s)
    TIME_SINCE_LAST_SCALE=$((CURRENT_TIME - LAST_SCALE_TIME))

    log "Queue: $PENDING pending | $ACTIVE active | $CURRENT workers | Load: $TOTAL_LOAD"

    # Cooldown check
    if [ $TIME_SINCE_LAST_SCALE -lt $COOLDOWN ]; then
        COOLDOWN_REMAINING=$((COOLDOWN - TIME_SINCE_LAST_SCALE))
        # Uncomment for verbose cooldown logging:
        # log "Cooldown active (${COOLDOWN_REMAINING}s remaining)"
        sleep $CHECK_INTERVAL
        continue
    fi

    # Scale up logic
    if [ $TOTAL_LOAD -ge $SCALE_UP_THRESHOLD ] && [ $CURRENT -lt $MAX_WORKERS ]; then
        NEW_COUNT=$((CURRENT + 1))
        log "⬆️ Scaling UP: $CURRENT → $NEW_COUNT workers (load: $TOTAL_LOAD)"
        scale_workers $NEW_COUNT
        LAST_SCALE_TIME=$CURRENT_TIME

    # Scale down logic
    elif [ $TOTAL_LOAD -le $SCALE_DOWN_THRESHOLD ] && [ $CURRENT -gt $MIN_WORKERS ]; then
        NEW_COUNT=$((CURRENT - 1))
        log "⬇️ Scaling DOWN: $CURRENT → $NEW_COUNT workers (load: $TOTAL_LOAD)"
        scale_workers $NEW_COUNT
        LAST_SCALE_TIME=$CURRENT_TIME
    fi

    sleep $CHECK_INTERVAL
done
👉 Save as `autoscale.sh` in `/opt/n8n-queue-mode/`. Make executable: `chmod +x autoscale.sh`

Running as a Systemd Service

# Create systemd service file
sudo tee /etc/systemd/system/n8n-autoscale.service > /dev/null << 'EOF'
[Unit]
Description=N8N Worker Auto-Scaler
After=docker.service
Requires=docker.service
After=n8n-main.service

[Service]
Type=simple
WorkingDirectory=/opt/n8n-queue-mode
ExecStart=/opt/n8n-queue-mode/autoscale.sh
Restart=always
RestartSec=5
StandardOutput=journal
StandardError=journal
User=root

[Install]
WantedBy=multi-user.target
EOF

# Enable and start
sudo systemctl daemon-reload
sudo systemctl enable n8n-autoscale
sudo systemctl start n8n-autoscale

# Monitor
journalctl -u n8n-autoscale -f

📊 Auto-Scaler Performance:

  • Scale-up latency: ~30 seconds (detects load, scales, new workers ready)
  • Scale-down latency: ~60 seconds (cooldown prevents rapid oscillation)
  • Max workers reached: 4 (for 500 alerts, 4-5 is optimal)
  • CPU utilization: 60-85% during peak, 15-20% at baseline
  • Memory per worker: ~120MB (2 workers = 240MB, 4 workers = 480MB)

Monitoring: Real-Time Visibility

Key Metrics to Monitor

Metric Source Alert Threshold Why It Matters
Redis Queue Depth LLEN bull:jobs:wait >50 for 2min Indicates workflow backlog. Growing queue = need more workers or workflow optimization.
Postgres Connections SELECT count(*) FROM pg_stat_activity >20 Too many connections indicate bottlenecks. Postgres default is 100 max.
Worker Count docker compose ps Monitor scaling Verify auto-scaler is working. Should scale up under load, down when idle.
Execution Success Rate Postgres: SELECT status, COUNT(*) FROM execution_entity GROUP BY status <95% success Failures indicate workflow errors, missing credentials, or infrastructure issues.
Avg Execution Time Postgres: SELECT AVG(EXTRACT(EPOCH FROM (stoppedAt - startedAt))) FROM execution_entity Baseline + 50% Slow executions indicate bottlenecks (Postgres slow, API timeouts, etc.)
Postgres Disk Usage SELECT pg_size_pretty(pg_database_size('n8n_queue')) >10GB Execution logs bloat disk. Pruning should keep this <5GB with retention=7 days.

Best Practices from Testing

  • Always load test before production — Use at least 300+ concurrent alerts to expose bottlenecks
  • Test with realistic workflows — Our whitelist check + Postgres + email + Jira = 15 nodes. Simpler test workflows won't reveal issues
  • Monitor auto-scaler cooldown — Without cooldown, workers scale up/down every 10s wasting resources
  • Set execution pruning aggressively — 7-day retention prevents Postgres bloat. Adjust based on compliance needs
  • Over-provision slightly — Run at 60-80% max capacity. Reserve 20-40% for spikes
  • Test failure scenarios — What happens if Postgres goes down? Redis crashes? A worker pod dies?
